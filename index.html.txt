function onBook(id,sel){
  const b=BOOKS.find(x=>x.code===sel.value);
  const item=mItems.find(i=>i.id===id);
  if(!item)return;
  if(b){item.code=b.code;item.name=b.name;item.price=b.price;item.dis=b.dis;}
  renderItems();
}

function onItemF(id,f,v){const item=mItems.find(i=>i.id===id);if(item){item[f]=Number(v)||0;recalc();updatePrev();}}

function iNet(item){return Math.round(item.qty*item.price*(1-item.dis/100));}

function recalc(){
  const sub=mItems.reduce((s,i)=>s+i.qty*i.price,0);
  const ex=Number(document.getElementById('cm-extra').value)||0;
  const discTotal=mItems.reduce((s,i)=>s+i.qty*i.price*i.dis/100,0);
  const net=Math.round(sub-discTotal-ex);
  const paid=Number(document.getElementById('cm-paid').value)||0;
  const due=Math.max(0,net-paid);
  document.getElementById('cm-subtotal').value=sub;
  document.getElementById('cm-net').value=net;
  document.getElementById('cm-due').value=due;
  document.getElementById('item-cnt').textContent=mItems.length;
  document.getElementById('total-qty').textContent=mItems.reduce((s,i)=>s+(Number(i.qty)||0),0);
  updatePrev();
}

// Bengali amount in words
function numWords(n){
  const u=['','এক','দুই','তিন','চার','পাঁচ','ছয়','সাত','আট','নয়','দশ','এগারো','বারো','তেরো','চোদ্দ','পনেরো','ষোলো','সতেরো','আঠারো','উনিশ','বিশ'];
  const t=['','','বিশ','ত্রিশ','চল্লিশ','পঞ্চাশ','ষাট','সত্তর','আশি','নব্বই'];
  if(n<=0)return'শূন্য';
  let w='';
  if(n>=10000000){w+=numWords(Math.floor(n/10000000))+' কোটি ';n%=10000000;}
  if(n>=100000){w+=numWords(Math.floor(n/100000))+' লক্ষ ';n%=100000;}
  if(n>=1000){w+=numWords(Math.floor(n/1000))+' হাজার ';n%=1000;}
  if(n>=100){w+=numWords(Math.floor(n/100))+' শত ';n%=100;}
  if(n>0){w+=n<=20?u[n]:t[Math.floor(n/10)]+(n%10?' '+u[n%10]:'');}
  return w.trim()+' টাকা মাত্র';
}
